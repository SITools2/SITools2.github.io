/*global Ext, ID*/
function getI(c) {
    return c.getId();
}
function getCmp(id) {
    return Ext.getCmp(id);
}

function getMenu() {
    return Ext.getCmp(ID.CMP.MENU);
}

function getMenuRoot() {
    return getMenu().getRootNode();
}

function getMenuNode(id) {
    return getMenu().getNodeById(id);
}
