        Ext.apply(config, {

                method : 'GET'

            idProperty : 'id',
            fields : [ {
                name : 'id',
                type : 'string'
            }, {
                name : 'name',
                type : 'string'
            }, {
                name : 'description',
                type : 'string'
            }, {
                name : 'label',
                type : 'string'
            }, {
                name : 'xtype',
                type : 'string'
            }, {
                name : 'author',
                type : 'string'
            }, {
                name : 'version',
                type : 'string'
            }, {
                name : 'icon',
                type : 'string'
            }, {
                name : 'parameters'                    
            }, {
                name : 'priority',
                type : 'int'
            }, {
                name : 'dataSetSelection'
            }, {
                name : 'dependencies'
            }, {
                name : 'defaultVisibility'
            }, {
                name : 'parametersMap'
            }]
        });
        
        sitools.user.component.dataviews.services.GuiServicesStore.superclass.constructor.call(this, config);
     }
});
