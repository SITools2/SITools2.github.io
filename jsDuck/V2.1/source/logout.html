/*global Ext,window*/
function utils_logout() {

	Ext.util.Cookies.set('userLogin', '');
	Ext.util.Cookies.set('hashCode', '');
	Ext.Ajax.defaultHeaders.Authorization = '';
	window.location.reload();
}
